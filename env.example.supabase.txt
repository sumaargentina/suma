# Supabase Configuration
# Obtén estas credenciales de tu proyecto en https://supabase.com
# Settings > API

NEXT_PUBLIC_SUPABASE_URL=https://your-project-url.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=your-anon-key-here
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# =====================================================
# IMPORTANTE: Copia este archivo a .env y reemplaza
# los valores con tus credenciales reales de Supabase
# =====================================================

# Notas:
# - NEXT_PUBLIC_SUPABASE_URL: URL de tu proyecto (ej: https://abcdefg.supabase.co)
# - NEXT_PUBLIC_SUPABASE_ANON_KEY: Clave pública (segura para usar en el cliente)
# - SUPABASE_SERVICE_ROLE_KEY: Clave privada (SOLO para uso en el servidor, NUNCA la expongas en el cliente)
